cmake_minimum_required(VERSION 4.0)
project(CMakeVulkan)

set(EXEC_TARGET AppPrj)
file(GLOB_RECURSE SOURCES ${CMAKE_SOURCE_DIR}/src/*.cpp)
file(GLOB_RECURSE HEADERS ${CMAKE_SOURCE_DIR}/src/*.h ${CMAKE_SOURCE_DIR}/include/*.h)

add_executable(${EXEC_TARGET} ${HEADERS} ${SOURCES})
set_property(TARGET ${EXEC_TARGET} PROPERTY CXX_STANDARD 20)
target_include_directories(${EXEC_TARGET} PRIVATE include src)

# _DEBUG macro for debug builds (enables Vulkan validation layers)
target_compile_definitions(${EXEC_TARGET} PRIVATE 
  $<$<CONFIG:Debug>:_DEBUG>
  # depth range: OpenGL uses [-1, 1], Vulkan uses [0, 1]
  GLM_FORCE_DEPTH_ZERO_TO_ONE
)

# MSVC: enable conformant preprocessor (required for __VA_OPT__)
if(MSVC)
    target_compile_options(${EXEC_TARGET} PRIVATE /Zc:preprocessor)
endif()

# Library linking and Headers import
find_package(Vulkan REQUIRED)
find_package(glfw3 REQUIRED)
find_package(tinyobjloader REQUIRED)
find_package(glm REQUIRED)
find_package(stb REQUIRED)

target_link_libraries(${EXEC_TARGET} PRIVATE
  Vulkan::Vulkan
  glfw
  tinyobjloader::tinyobjloader
  glm::glm
  stb::stb
)

file(GLOB_RECURSE SHADERS ${CMAKE_SOURCE_DIR}/resources/shaders/*.vert ${CMAKE_SOURCE_DIR}/resources/shaders/*.frag)
foreach(SHADER ${SHADERS})
  get_filename_component(SHADER_NAME ${SHADER} NAME)   # triangle.vert
  set(SPV_OUTPUT ${CMAKE_SOURCE_DIR}/resources/shaders/${SHADER_NAME}.spv) # <full-path>/triangle.vert.spv
  add_custom_command(
    OUTPUT ${SPV_OUTPUT}
    COMMAND Vulkan::glslc ${SHADER} -o ${SPV_OUTPUT}
    DEPENDS ${SHADER}
    COMMENT "Compiling ${SHADER_NAME} -> ${SHADER_NAME}.spv"
  )
  list(APPEND SPV_OUTPUTS ${SPV_OUTPUT})
endforeach(SHADER ${SHADERS})
add_custom_target(Shaders DEPENDS ${SPV_OUTPUTS})
add_dependencies(${EXEC_TARGET} Shaders)

# Post build: Copy Resources to bin
set(ResourcesPath ${CMAKE_CURRENT_SOURCE_DIR}/resources)
add_custom_command(TARGET ${EXEC_TARGET} POST_BUILD
  COMMAND ${CMAKE_COMMAND} -E cmake_echo_color --cyan "Copy Resources..."
  COMMAND ${CMAKE_COMMAND} -E copy_directory ${ResourcesPath} $<TARGET_FILE_DIR:${EXEC_TARGET}>/resources
)
