cmake_minimum_required(VERSION 4.0)
project(CMakeVulkan)


set(EXEC_TARGET AppPrj)
file(GLOB_RECURSE SOURCES ${CMAKE_SOURCE_DIR}/src/*.cpp)
file(GLOB_RECURSE HEADERS ${CMAKE_SOURCE_DIR}/src/*.h ${CMAKE_SOURCE_DIR}/include/*.h)

add_executable(${EXEC_TARGET} ${HEADERS} ${SOURCES})
set_property(TARGET ${EXEC_TARGET} PROPERTY CXX_STANDARD 20)
target_include_directories(${EXEC_TARGET} PRIVATE include src)

# _DEBUG macro for debug builds (enables Vulkan validation layers)
target_compile_definitions(${EXEC_TARGET} PRIVATE 
  $<$<CONFIG:Debug>:_DEBUG>
  # depth range: OpenGL uses [-1, 1], Vulkan uses [0, 1]
  GLM_FORCE_DEPTH_ZERO_TO_ONE
)

# MSVC: enable conformant preprocessor (required for __VA_OPT__)
if(MSVC)
    target_compile_options(${EXEC_TARGET} PRIVATE /Zc:preprocessor)
endif()

#fix on web
set(CMAKE_MAP_IMPORTED_CONFIG_DEBUG Release)
set(CMAKE_MAP_IMPORTED_CONFIG_RELWITHDEBINFO Release)
set(CMAKE_MAP_IMPORTED_CONFIG_MINSIZEREL Release)

# Library linking and Headers import
find_package(Vulkan REQUIRED)
find_package(glfw3 REQUIRED)
find_package(tinyobjloader REQUIRED)
find_package(glm REQUIRED)

target_link_libraries(${EXEC_TARGET} PRIVATE
  Vulkan::Vulkan
  glfw
  tinyobjloader::tinyobjloader
  glm::glm
)

set(SHADER_SOURCE ${CMAKE_SOURCE_DIR}/resources/shaders/triangle.glsl)

add_custom_command(
    OUTPUT ${CMAKE_SOURCE_DIR}/resources/shaders/triangle.vert.spv
    COMMAND Vulkan::glslc
            -DVERTEX_SHADER
            -fshader-stage=vert
            ${SHADER_SOURCE}
            -o ${CMAKE_SOURCE_DIR}/resources/shaders/triangle.vert.spv
    DEPENDS ${SHADER_SOURCE}
    COMMENT "Compiling vertex shader"
)

add_custom_command(
    OUTPUT ${CMAKE_SOURCE_DIR}/resources/shaders/triangle.frag.spv
    COMMAND Vulkan::glslc
            -DFRAGMENT_SHADER
            -fshader-stage=frag
            ${SHADER_SOURCE}
            -o ${CMAKE_SOURCE_DIR}/resources/shaders/triangle.frag.spv
    DEPENDS ${SHADER_SOURCE}
    COMMENT "Compiling fragment shader"
)

add_custom_target(Shaders
    DEPENDS
        ${CMAKE_SOURCE_DIR}/resources/shaders/triangle.vert.spv
        ${CMAKE_SOURCE_DIR}/resources/shaders/triangle.frag.spv
)

add_dependencies(${EXEC_TARGET} Shaders)

# Post build: Copy Resources to bin
set(ResourcesPath ${CMAKE_CURRENT_SOURCE_DIR}/resources)
add_custom_command(TARGET ${EXEC_TARGET} POST_BUILD
  COMMAND ${CMAKE_COMMAND} -E cmake_echo_color --cyan "Copy Resources..."
  COMMAND ${CMAKE_COMMAND} -E copy_directory ${ResourcesPath} $<TARGET_FILE_DIR:${EXEC_TARGET}>/resources
)
